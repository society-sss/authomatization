    <script>
        document.getElementById('menu-toggle').addEventListener('click', function() {
            const menu = document.getElementById('mobile-menu');
            menu.classList.toggle('hidden');
        });
        
    const form = document.getElementById('form');
    const privacyCheckbox = document.getElementById('privacy');
    const privacyError = document.getElementById('privacy-error');
    const submitBtn = document.getElementById('submit-btn');

    if (form) {
        privacyCheckbox.addEventListener('change', function() {
            if(this.checked) {
                privacyError.classList.add('hidden');
                submitBtn.classList.remove('opacity-50', 'cursor-not-allowed');
                submitBtn.classList.add('hover:bg-orange-600');
            } else {
                submitBtn.classList.add('opacity-50', 'cursor-not-allowed');
                submitBtn.classList.remove('hover:bg-orange-600');
            }
        });

        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            if(!privacyCheckbox.checked) {
                privacyError.classList.remove('hidden');
                submitBtn.classList.add('opacity-50', 'cursor-not-allowed');
                submitBtn.classList.remove('hover:bg-orange-600');
                return;
            }
            
            window.location.href = './pages/thank/thank.html';
        });
    }



        function toggleFAQ(id) {
            const answer = document.getElementById(`faq-answer-${id}`);
            const icon = document.getElementById(`faq-icon-${id}`);
            
            answer.classList.toggle('hidden');
            icon.classList.toggle('transform');
            icon.classList.toggle('rotate-180');
        }

        if (!localStorage.getItem('cookiesAccepted')) {
            document.getElementById('cookie-banner').classList.remove('hidden');
        }

        document.getElementById('accept-cookies').addEventListener('click', function() {
            localStorage.setItem('cookiesAccepted', 'true');
            document.getElementById('cookie-banner').classList.add('hidden');
        });

        document.getElementById('reject-cookies').addEventListener('click', function() {
            localStorage.setItem('cookiesAccepted', 'false');
            document.getElementById('cookie-banner').classList.add('hidden');
        });

        const fadeElements = document.querySelectorAll('.fade-in');
        
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('opacity-100');
                    observer.unobserve(entry.target);
                }
            });
        }, {
            threshold: 0.1
        });

        fadeElements.forEach(element => {
            element.classList.add('opacity-0');
            observer.observe(element);
        });
    </script>
